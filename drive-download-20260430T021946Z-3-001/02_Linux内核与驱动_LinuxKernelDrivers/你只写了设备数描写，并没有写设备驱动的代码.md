在 Linux 内核中，编写 I2C 驱动通常遵循 i2c_driver 框架。以下是一个典型的 RK3568 驱动模板，它展示了如何与刚才提到的设备树节点进行匹配，并读取外设信息。

### 1. 核心驱动代码实现

\#include \<linux/module.h\>\
\#include \<linux/i2c.h\>\
\#include \<linux/of.h\>\
\#include \<linux/uaccess.h\>\
\
/\* 1. 定义设备私有结构体 \*/\
struct my_i2c_dev {\
struct i2c_client \*client;\
int data;\
};\
\
/\* 2. 读取外设寄存器的函数示例 \*/\
static int my_i2c_read_reg(struct i2c_client \*client, u8 reg)\
{\
/\* 使用 SMBus 接口读取一个字节，这是最推荐的方式 \*/\
int val = i2c_smbus_read_byte_data(client, reg);\
if (val \< 0) {\
dev_err(&client-\>dev, "Read reg 0x%x failed\n", reg);\
}\
return val;\
}\
\
/\* 3. Probe 函数：设备树匹配成功后执行 \*/\
static int my_i2c_probe(struct i2c_client \*client, const struct i2c_device_id \*id)\
{\
struct my_i2c_dev \*my_dev;\
int ret;\
\
dev_info(&client-\>dev, "My I2C Device Probed!\n");\
\
/\* 为私有数据分配内存 \*/\
my_dev = devm_kzalloc(&client-\>dev, sizeof(struct my_i2c_dev), GFP_KERNEL);\
if (!my_dev)\
return -ENOMEM;\
\
my_dev-\>client = client;\
i2c_set_clientdata(client, my_dev);\
\
/\* 示例：读取外设的 ID 寄存器（假设地址为 0x00） \*/\
ret = my_i2c_read_reg(client, 0x00);\
if (ret \>= 0) {\
dev_info(&client-\>dev, "Device ID: 0x%02x\n", ret);\
}\
\
return 0;\
}\
\
/\* 4. Remove 函数：设备卸载时执行 \*/\
static int my_i2c_remove(struct i2c_client \*client)\
{\
dev_info(&client-\>dev, "My I2C Device Removed\n");\
return 0;\
}\
\
/\* 5. 设备树匹配表 (必须与 DTS 中的 compatible 属性完全一致) \*/\
static const struct of_device_id my_i2c_of_match\[\] = {\
{ .compatible = "vendor,model_name", },\
{ /\* sentinel \*/ }\
};\
MODULE_DEVICE_TABLE(of, my_i2c_of_match);\
\
/\* 6. i2c_driver 结构体 \*/\
static struct i2c_driver my_i2c_driver = {\
.driver = {\
.name = "my_i2c_driver",\
.of_match_table = my_i2c_of_match,\
},\
.probe = my_i2c_probe,\
.remove = my_i2c_remove,\
};\
\
/\* 7. 驱动模块入口与出口 \*/\
module_i2c_driver(my_i2c_driver);\
\
MODULE_LICENSE("GPL");\
MODULE_AUTHOR("Engineer");\
MODULE_DESCRIPTION("RK3568 I2C Client Driver Example");

### 2. 代码逻辑解析

#### 匹配机制

驱动中的 of_match_table 里的 compatible 字符串 "vendor,model_name" 是连接 DTS 和内核代码的“胶水”。当内核加载时，它会发现总线上有一个地址为 0x50 的设备，并根据其 compatible 属性找到对应的驱动并调用 probe。

#### 数据读取接口的选择

- **i2c_smbus_read_byte_data**: 代码中使用了这个接口。它是对底层 I2C 读写时序的封装。对于 RK3568 这种标准的 I2C 控制器，SMBus 系列函数是最稳定、最易读的选择。

- **i2c_transfer**: 如果你的外设需要非标准的通信时序（例如一次性读取 100 个非连续字节），你需要自己构造 struct i2c_msg 并调用 i2c_transfer。

#### 资源管理

使用了 devm_kzalloc。这是内核提供的“受管理”的内存分配。优点是当驱动卸载或 probe 失败时，内核会自动释放内存，避免内存泄漏，这对嵌入式系统的长期稳定性至关重要。

### 3. 编译与测试建议

1.  **Makefile**: 确保 Makefile 中的 obj-m 指向你的源文件。

2.  **加载**: 编译成 .ko 文件后，使用 insmod my_i2c.ko 加载。

3.  **日志**: 观察 dmesg 输出。如果看到 Device ID: 0xXX，说明你已经成功通过 I2C 子系统读取到了 RK3568 挂载的外设信息。
